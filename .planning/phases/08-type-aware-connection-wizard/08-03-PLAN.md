```lua
local record = submission.params
if source_is_filesource(source_meta) and is_scoped_wizard_mode(submission.wizard.mode) then

    wizard = submission.wizard,
  })
elseif source_is_filesource(source_meta) and conn_id ~= nil then
  record = vim.tbl_extend("force", submission.params, {
    __remove_keys = { "wizard" },
  })
```
